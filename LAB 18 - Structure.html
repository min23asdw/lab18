    #include<iostream>
    using namespace std;
    struct Rect{
    double x,y,w,h;
    };
    double overlap(Rect R1,Rect R2){
    double y;
    double x;
    y = min( R1.y , R2.y ) - max( R1.y-R1.h , R2.y-R2.h );
    x = min( R1.x + R1.w , R2.x+R2.w ) - max( R1.x , R2.x);
    if(x*y<=0){return 0;}else{
    return x*y;
    }
    }

    • [Line 67 ใน Source Code เริ่มต้น] โปรแกรมในส่วนที่ให้มาจะอ่านข้อมูลของนักศึกษาแต่ละคนที่อยู่ในไฟล์ students.txt ไว้ในตัวแปร textline ขอให้นักศึกษาแยกข้อมูลใน textline ออกเป็นชื่อ รหัสนักศึกษา เพศ และเกรดเฉลี่ย และเก็บข้อมูลที่แยกแล้วไว้ในตัวแปร s เพื่อที่โปรแกรมให้ในส่วนต่อไปจะรวบรวมข้อมูลของนักศึกษาแต่ละคนในตัวแปร s ไปเก็บไว้ในตัวแปร allstudents ซึ่งเป็น vector ของ struct student
    • [Line 87 ใน Source Code เริ่มต้น] โปรแกรมในส่วนนี้จะเป็นการอ่านข้อมูลผู้สอนในแต่ละรายวิชาในไฟล์ courses.txt ซึ่งชื่อของผู้สอนแต่ละคนจะถูกเก็บไว้ในตัวแปร textline ขอให้นักศึกษานำชื่อผู้สอนที่อยู่ในตัวแปร textline นี้ไปเพิ่มไว้ที่ data member lecture_list[] ของ struct course ของวิชาล่าสุดที่อยู่ในตัวแปร allcourses[] 
    • [Line 23 ใน Source Code เริ่มต้น] แก้ไข Function Signature student * findstudent(vector<student> allstudents,int key)เพื่อให้ทำงานได้อย่างถูกต้อง
    • [Line 96 ใน Source Code เริ่มต้น] โปรแกรมในส่วนนี้จะเป็นการอ่านข้อมูลผู้เรียนในแต่ละรายวิชาในไฟล์ courses.txt ซึ่ง pointer ที่ชี้ไปถึงผู้เรียนแต่ละคนจะอยู่ในตัวแปร p ขอให้นักศึกษานำ pointer p นี้ไปเพิ่มไว้ที่ data member student_list[] ของ struct course ของวิชาล่าสุดที่อยู่ในตัวแปร allcourses[]
